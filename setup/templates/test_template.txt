import os
import sys
from main.day{day:02d}.{puzzle_name} import solve


def test_simple():
    f = open(os.path.join(sys.path[0], "data/test_input.txt"))
    lines = []
    for line in f:
        lines.append(int(line.strip('\n')))
    assert solve(1, 2) == 3
